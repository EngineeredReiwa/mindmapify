Add Node Test

Open browser console and run:

// Check if store is available
console.log('Store available:', !!window.mindmapStore);

// Get current state
const state = window.mindmapStore?.getState();
console.log('Current nodes:', state?.nodes);

// Try to add a node
state?.addNode({ x: 100, y: 100 }, 'Test Node');
console.log('After adding:', state?.nodes);

// Check toolbar button
const addButton = document.querySelector('button[title*="Add"]');
console.log('Add button found:', !!addButton);
addButton?.click();
